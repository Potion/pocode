05945         inline void SetHandle(XnContext* pContext)
05946         {
05947             if (m_pContext == pContext)
05948             {
05949                 return;
05950             }
05951 
05952             if (m_pContext != NULL)
05953             {
05954                 if (m_bUsingDeprecatedAPI && m_bAllocated)
05955                 {
05956                     // Backwards compatibility: call shutdown instead of release, to make old programs get the 
05957                     // exact same behavior they used to have.
05958                     xnForceShutdown(m_pContext);
05959                 }
05960                 else
05961                 {
05962                     xnContextUnregisterFromShutdown(m_pContext, m_hShuttingDownCallback);
05963                     xnContextRelease(m_pContext);
05964                 }
05965             }
05966 
05967             if (pContext != NULL)
05968             {
05969                 XnStatus nRetVal = xnContextAddRef(pContext);
05970                 XN_ASSERT(nRetVal == XN_STATUS_OK);
05971 
05972                 nRetVal = xnContextRegisterForShutdown(pContext, ContextShuttingDownCallback, this, &m_hShuttingDownCallback);
05973                 XN_ASSERT(nRetVal == XN_STATUS_OK);
05974             }
05975 
05976             m_pContext = pContext;
05977         }
05978 
05979         inline void TakeOwnership(XnContext* pContext)
05980         {
05981             SetHandle(pContext);
05982 
05983             if (pContext != NULL)
05984             {
05985                 xnContextRelease(pContext);
05986             }
05987         }
